vim run-gunicorn.sh
	#!/usr/bin/bash
	cd /home/osboxes/RPL/rpl/back/pbmps-api
	/home/osboxes/RPL/rpl/back/pbmps-api/bin/gunicorn --workers 3 --bind 0.0.0.0:4000 -m 007 wsgi:app

vim /etc/systemd/system/pbmps-api-server.service
[Unit]
Description=Gunicorn instance to serve api-server
After=network.target

[Service]
#Service to specify the user and group under which our process will run
User=osboxes

# give group ownership to the www-data group so that Nginx can communicate easily with the Gunicorn processes
Group=nginx

# specify working directory and env variables
WorkingDirectory=/home/osboxes/RPL/rpl/back/pbmps-api
Environment="PATH=/home/osboxes/RPL/rpl/back/pbmps-api/bin"

#command to start the service
ExecStart=/usr/bin/bash /home/osboxes/RPL/rpl/back/pbmps-api/run-gunicorn.sh

#start the service when regular multi-user system is up and running
[Install]
WantedBy=multi-user.target

systemctl start pbmps-api-server.service
